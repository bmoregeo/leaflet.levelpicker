Default

var defaultMap = L.map('defaultMap', {
    center: coordinates,
    zoom: zoom,
    layers: [L.tileLayer(mapboxUrl, {id: 'examples.map-20v6611k'})],
    zoomControl: false,
    dragging: true
});

new L.Control.LevelPicker({levels:levels}).addTo(defaultMap);

defaultMap.on('level.change', function(e) {
    alert('Change to Level ' + levels[e.target.level].level);
});
